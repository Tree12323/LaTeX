	\begin{gather}
		a + b = b + a \\
		ab = ba
	\end{gather}
	
	\begin{gather*}
	a + b = b + a \\
	ab = ba
	\end{gather*}
	% 在 \\ 前加入 \notag 阻止编号
	\begin{gather}
	a + b = b + a \notag \\
	ab = ba \notag \\
	a^2 + b^2 = 1
	\end{gather}
	
	\begin{align}
		x &= t + \cos t + 1 \notag \\
		y &= 2\sin t
	\end{align}
	
	\begin{align*}
		x &= t & x &= \cos t & x &=t \\
		y &= 2t & y &= \sin (t + 1) & y &= \sin t
	\end{align*}
	
	% split 环境（对齐采用align环境的方式，编号在中间）
	\begin{equation}
		\begin{split}
			\cos 2x &= \cos^2 x - \sin^2 x \\
			&= 2\cos^2 x - 1
		\end{split}
	\end{equation}
	
%	 case 环境
%	 每行使用 & 分隔两部分，
%	 通常表示值和后面的条件

		D(s) = \begin{cases}
		1, & \text{如果 } x \in \mathbb{Q}; \\
		0, & \text{如果 } x \in \mathbb{R} \setminus\mathbb{Q}
		\end{cases}
	\end{equation}
