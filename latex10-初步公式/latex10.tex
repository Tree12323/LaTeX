	\section{行间公式}
	\subsection{美元符号}
	交换律是
	$$a+b=b+a$$
	如 $$1+2=2+1$$。
	\subsection{中括号}
	交换律是
	\[a+b=b+a\]
	如 \[1+2=2+1\]。
	\subsection{displaymath 环境}
	交换律是 \begin{displaymath}a+b=b+a\end{displaymath}
	如 \begin{displaymath}1+2=2+1\end{displaymath}。
	\subsection{自动编号公式equation环境}
	交换律见式 \ref{eq:commuative}

	\end{equation}
